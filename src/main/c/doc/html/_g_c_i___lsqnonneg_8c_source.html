159 void Householder(int mode, double *v, int p, int l, int m, double *u_p,
160  double *C[], int nC)
161 {
162  double vmax, vmax_inv, vtmp, s, b, tj;
163  double *cj;
164  int i, j;
165 
166  if (0>p || p>=l || l>m) return;
167 
168  /* find the maximum v_i */
169  vmax = fabs(v[p]);
170  if (mode == 1) {
171  for (i=l; i<m; i++)
172  vmax = max(vmax, fabs(v[i]));
173  if (vmax == 0) return; /* oops; not much to do here, really */
174  vmax_inv = 1/vmax; /* only calculate this once; does it matter? */
175 
176  /* we now calculate s, which is step 1 in the algorithm */
177  vtmp = v[p]*vmax_inv;
178  s = vtmp*vtmp;
179  for (i=l; i<m; i++) {
180  vtmp = v[i]*vmax_inv;
181  s += vtmp*vtmp;
182  }
183  s = vmax*sqrt(s);
184  /* this is step 1 done */
185  if (v[p]>0) s = -s;
186  *u_p = v[p]-s;
187  v[p] = s;
188  } else {
189  if (vmax == 0) return; /* same test as before, only now we are
190  testing only the v_p element, which is
191  actually y_p */
192  }
193 
194  /* Now calculate the Qc_j vectors */
195  if (nC <= 0 || C==NULL) return;
196  b = v[p] * (*u_p);
197  if (b >= 0) return;
198 
199  for (j=0; j<nC; j++) {
200  cj = C[j];
201  tj = cj[p] * (*u_p);
202  for (i=l; i<m; i++)
203  tj += cj[i]*v[i];
204 
205  if (tj != 0) {
206  tj /= b;
207  cj[p] += tj*(*u_p);
208  for (i=l; i<m; i++)
209  cj[i] += tj*v[i];
210  }
211  }
212 }

241 void GivensCalc(double v_1, double v_2, double *c, double *s, double *r)
242 {
243  double w, q;
244 
245  if (fabs(v_1) > fabs(v_2)) {
246  w = v_2/v_1;
247  q = sqrt(1+w*w);
248  *c = 1/q;
249  if (v_1 < 0) *c = - (*c);
250  *s = w*(*c);
251  *r = fabs(v_1)*q;
252  } else {
253  if (v_2 == 0) {
254  *c = 1; *s = 0; *r = 0;
255  } else {
256  w = v_1/v_2;
257  q = sqrt(1+w*w);
258  *s = 1/q;
259  if (v_2 < 0) *s = - (*s);
260  *c = w*(*s);
261  *r = fabs(v_2)*q;
262  }
263  }
264 }

388 int GCI_lsqnonneg(double **A_orig, double *b_orig, double *x, int m, int n,
389  int preserve, double *rnorm_orig, double *lambda)
390 {
391  double *AA[MAX_VARS], bb[MAX_EQNS], **A, *b; /* to preserve originals */
392  double rnorm2, *rnorm; /* in case rnorm_orig is NULL */
393  double ww[MAX_VARS], z[MAX_EQNS], *w;
394  double *C[MAX_VARS]; /* for passing to Householder */
395  double wmax, asave, u_p, unorm, sum; /* miscellany */
396  int iter, itmax;
397  int index[MAX_VARS], p;
398  int i, ii, j, jj, k, l, q, zmax, index_zmax;
399 
400  /* Meanings of variables:
401  index[] is an array containing information on the sets Z and P.
402  index[0..p-1] is the set P
403  index[p..n-1] is the set Z
404 
405  z[] is working space
406  */
407 
408  /* Check variables */
409  if (m > MAX_EQNS || m <= 0) return -1;
410  if (n > MAX_VARS || n <= 0) return -2;
411  w = (lambda == NULL) ? ww : lambda;
412  rnorm = (rnorm_orig == NULL) ? &rnorm2 : rnorm_orig;
413  if (preserve) {
414  /* We allocate one long array and split it into pieces */
415  if ((AA[0] = (double *)malloc((unsigned)(m*n)*sizeof(double))) == NULL)
416  return -4;
417  for (i=0; i<n; i++) {
418  AA[i] = AA[0]+m*i;
419  for (j=0; j<m; j++)
420  AA[i][j] = A_orig[i][j];
421  }
422  for (j=0; j<m; j++)
423  bb[j] = b_orig[j];
424  A = AA;
425  b = bb;
426  } else {
427  A = A_orig;
428  b = b_orig;
429  }
430 
431  iter = 0;
432  itmax = 3*n;
433 
434  /* Initialise the index and x arrays */
435  p = 0;
436  for (i=0; i<n; i++) {
437  x[i] = 0;
438  index[i] = i;
439  }
440 
441  /* Main loop */
442  while (p<n && p<m) {
443  /* Compute the dual (negative gradient) vector w=A^T(b-Ax)
444  We are only interested in the components of this vector in Z,
445  so we only calculate these. It turns out that for these
446  components, we have w_j=(A^T.b)_j, and that (A^T.Ax)_j=0,
447  remembering that A is continually modified. I really don't
448  yet know why this is true. */
449 
450  for (i=p; i<n; i++) {
451  j = index[i];
452  sum = 0;
453  for (k=p; k<m; k++)
454  sum += Amatrix(k,j) * b[k];
455  w[j] = sum;
456  }
457 
458  wmax = 0;
459  while (wmax == 0) {
460  /* We now find the maximum positive w_j. */
461  zmax = -1;
462  for (i=p; i<n; i++) {
463  j = index[i];
464  if (w[j]>wmax) {
465  wmax = w[j];
466  zmax = i;
467  }
468  }
469 
470  if (wmax <= 0) /* all w_j non-positive, so done */
471  goto terminate;
472 
473  /* We move index[zmax] from Z to P
474  We begin the transformation and check the new diagonal element
475  to avoid near linear dependence */
476  index_zmax = index[zmax];
477  asave = Amatrix(p,index_zmax);
478 
479  Householder(1, A[index_zmax], p, p+1, m, &u_p, NULL, 0);
480  /* u has ended up in the index_zmax-th column of A */
481  unorm = 0;
482  for (k=0; k<p; k++)
483  unorm += Amatrix(k,index_zmax)*Amatrix(k,index_zmax);
484  unorm = sqrt(unorm);
485 
486  /* Is the index_zmax-th column sufficiently independent
487  (whatever that means)? */
488  if ((unorm != 0 && fabs(Amatrix(p,index_zmax)/unorm) < TOL) ||
489  (unorm == 0 && Amatrix(p,index_zmax) == 0)) {
490  /* no, it isn't */
491  Amatrix(p,index_zmax) = asave;
492  w[index_zmax] = 0;
493  wmax = 0;
494  } else {
495  /* it is: copy b[] into z[], update z[] and solve for ztest,
496  the proposed new value for x[index_zmax] */
497  double ztest;
498  for (i=0; i<m; i++)
499  z[i] = b[i];
500  C[0] = z; /* Need to pass Householder an array of pointers */
501  Householder(2, A[index_zmax], p, p+1, m, &u_p, C, 1);
502  ztest = z[p] / Amatrix(p,index_zmax);
503  if (ztest <= 0) {
504  /* nah, this won't do, try a different zmax */
505  Amatrix(p,index_zmax) = asave;
506  w[index_zmax] = 0;
507  wmax = 0;
508  }
509  /* We've now got an acceptable index to move from Z to P!
510  We'll leave this while (wmax==0) loop now, then. */
511  }
512  }
513 
514  /* And take heed of the fact that we've got our index */
515  for (i=0; i<m; i++)
516  b[i] = z[i];
517  index[zmax] = index[p];
518  index[p] = index_zmax;
519  p++;
520 
521  /* We now apply our Householder transformation to all of the
522  columns of A which correspond to index[p], ..., index[n-1]. We
523  could call the Householder function for each column separately,
524  but it's probably more efficient (fewer function calls) to
525  bundle the columns up into one new temporary array. Note that
526  we've incremented p! */
527  if (p<n) {
528  for (i=p; i<n; i++) {
529  C[i-p] = A[index[i]]; /* the index[i]-th column of A */
530  }
531  Householder(2, A[index_zmax], p-1, p, m, &u_p, C, n-p);
532  }
533 
534  /* ... zero the index_zmax-th column of A below the
535  diagonal, and set w[index_zmax]=0 ... */
536  for (i=p; i<m; i++)
537  Amatrix(i,index_zmax) = 0;
538  w[index_zmax] = 0;
539 
540  /* ... and solve the triangular system to solve the least squares
541  problem |A_P.z-b| (step 6 of algorithm), putting the solution
542  into z. Note also that a copy of b is still stored in z when
543  we begin.
544 
545  We have the matrix A_P which has j-th column being index[j],
546  j=0, ..., p-1. The j-th column is zero below the diagonal. So
547  we solve it by back-substitution. */
548 
549  for (i=p-1; i>=0; i--) {
550  j = index[i];
551  z[i] /= Amatrix(i,j);
552  for (k=0; k<i; k++)
553  z[k] -= Amatrix(k,j) * z[i];
554  }
555 
556  /* This next part is called the "secondary loop" in the book */
557  while (1) {
558  double alpha, tmp;
559 
560  if (++iter > itmax) /* over the iteration count, oops */
561  goto terminate;
562 
563  alpha = 2; /* greater than the values we will be
564  comparing, which will all be <= 1 */
565  for (i=0; i<p; i++)
566  if (z[i] <= 0) {
567  tmp = x[index[i]] / (x[index[i]]-z[i]); /* This is <= 1 */
568  if (tmp<alpha) {
569  alpha = tmp;
570  q = i; /* keep track of the minimum index */
571  }
572  }
573 
574  /* Were all z_j>0 for j in P? If so, set x:=z and break out of
575  this inner loop, returning to the main loop. */
576  if (alpha > 1) {
577  /* the next four lines of code corresponds to lines 244-247 in
578  the Fortran 66 version */
579  for (i=0; i<p; i++) {
580  x[index[i]] = z[i];
581  }
582  break;
583  }
584 
585  /* Interpolate between old x and new z using 0<alpha<=1 */
586  for (i=0; i<p; i++)
587  x[index[i]] += alpha * (z[i] - x[index[i]]);
588 
589  /* Now we have x[index[q]]=0, so modify A and b and the index
590  arrays to move index[q] and any other i with x[i]=0 from set
591  P to set Z */
592 
593  while (q >= 0) {
594  i = index[q];
595  x[i] = 0; /* ensure that rounding errors don't creep in */
596  /* now use Givens rotations to fix up A and b */
597  for (jj=q+1; jj<p; jj++) {
598  double c, s;
599  ii = index[jj];
600  index[jj-1] = ii;
601  GivensCalc(Amatrix(jj-1,ii), Amatrix(jj,ii),
602  &c, &s, &Amatrix(jj-1,ii));
603  Amatrix(jj,ii) = 0;
604 
605  for (l=0; l<n; l++)
606  if (l != ii)
607  GivensApply(c, s, &Amatrix(jj-1,l), &Amatrix(jj,l));
608  GivensApply(c, s, &b[jj-1], &b[jj]);
609  }
610  /* We've taken one element out of P */
611  p--;
612  index[p] = i;
613 
614  /* Check that the remaining coefficients in set P are
615  feasible, that is, >=0. They should be, because of
616  the way alpha was determined. If they are not,
617  this is due to rounding error, so we set any
618  non-positive elements to zero and move them too
619  from set P to set Z */
620  for (q=p-1; q>=0; q--)
621  if (x[index[q]] <= 0) break;
622  /* now if q>=0, then x[index[q]] <= 0,
623  and if q<0, we are done */
624  } /* end of while (q >= 0) loop */
625 
626  /* OK, all x[i]=0 have now had their indices moved from P
627  to Z. We can now solve the least squares problem
628  |Az-b| once again, ready for the next secondary loop.
629  So we simply copy b into z and solve the triangular
630  system, as before. We could reduce the code size a
631  fraction by having this code at the beginning of the
632  loop, but that would make the loop exit code (checking
633  iter) appear in the middle, which would be a tad
634  confusing. */
635 
636  for (i=0; i<m; i++)
637  z[i] = b[i];
638  for (i=p-1; i>=0; i--) {
639  j = index[i];
640  z[i] /= Amatrix(i,j);
641  for (k=0; k<i; k++)
642  z[k] -= Amatrix(k,j) * z[i];
643  }
644  } /* end of secondary loop */
645  } /* end of primary loop */
646 
647  terminate:
648  sum = 0;
649  if (p < m) {
650  for (i=p; i<m; i++)
651  sum += b[i] * b[i];
652  } else {
653  /* this is to get the w (dual) vector correct, in case we ever
654  decide to give it as an output to this function. */
655  for (i=0; i<n; i++)
656  w[i] = 0;
657  }
658 
659  (*rnorm) = sqrt(sum);
660 
661  if (preserve)
662  free(AA[0]);
663 
664  return (iter > itmax) ? -3 : 0;
665 }
